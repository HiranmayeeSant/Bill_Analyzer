/** * PSDR (Permanent Security Deposit Refund) Calculator * ---------------------------------------------------- * Fits alongside your existing Contract Demand / Power Factor savings modules. * * Expected input shape (adjust field names to match your existing extraction output): * * billData = { * supplyType: "HT" | "LT", * * // HT fields * ifPaidUpto: Number, // "If Paid Upto" amount from first page * lastMonthPayment: Number, // "Last Month Payment" amount from first page * * // LT fields * billingHistory: [ // Billing History table rows * { billAmount: Number, ... }, * ... * ], * * // Common * securityDepositHeld: Number // "Security Deposit Held (Rs.)" * } */ // Fixed Billing History window per MSEDCL bill format (LT consumers) const LT_BILLING_HISTORY_MONTHS = 13; // Returned whenever required fields are missing/unreadable — treated as // "no savings opportunity" rather than a hard error, since the consultant // workflow should never block on a single unreadable field. function noSavingsResult(supplyType, reason) { return { supplyType: supplyType || "UNKNOWN", securityDepositHeld: null, requiredDeposit: null, difference: null, psdrSavings: 0, recommendation: "No PSDR savings opportunity found.", dataIssue: reason, // internal/logging use — not shown to end user calculationBasis: {}, }; } function calculatePSDR(billData) { const { supplyType, securityDepositHeld } = billData; if (securityDepositHeld === undefined || securityDepositHeld === null) { return noSavingsResult(supplyType, "Security Deposit Held (Rs.) not found on bill."); } let requiredDeposit; // Y let calculationBasis = {}; if (supplyType === "HT") { const { ifPaidUpto, lastMonthPayment } = billData; if (ifPaidUpto === undefined || lastMonthPayment === undefined) { return noSavingsResult( supplyType, "'If Paid Upto' and/or 'Last Month Payment' not found on bill." ); } const greaterValue = Math.max(ifPaidUpto, lastMonthPayment); requiredDeposit = greaterValue * 2; calculationBasis = { ifPaidUpto, lastMonthPayment, greaterValue, formula: "Y = Greater(If Paid Upto, Last Month Payment) × 2", }; } else if (supplyType === "LT") { const { billingHistory } = billData; if (!Array.isArray(billingHistory) || billingHistory.length === 0) { return noSavingsResult(supplyType, "Billing History table not found or empty."); } // Use the most recent LT_BILLING_HISTORY_MONTHS rows if more are present; // if fewer are available, use whatever was extracted. const windowRows = billingHistory.slice(0, LT_BILLING_HISTORY_MONTHS); const amounts = windowRows .map((row) => row.billAmount) .filter((val) => typeof val === "number" && !isNaN(val)); if (amounts.length === 0) { return noSavingsResult( supplyType, "No valid Bill Amount values found in Billing History." ); } const averageBillAmount = amounts.reduce((sum, val) => sum + val, 0) / amounts.length; requiredDeposit = averageBillAmount * 2; calculationBasis = { monthsUsed: amounts.length, monthsExpected: LT_BILLING_HISTORY_MONTHS, averageBillAmount: round2(averageBillAmount), formula: `Y = Average Bill Amount (last ${amounts.length} months) × 2`, }; } else { return noSavingsResult(supplyType, `Unrecognized supply type: "${supplyType}".`); } const difference = securityDepositHeld - requiredDeposit; const hasSavings = difference > 0; return { supplyType, securityDepositHeld: round2(securityDepositHeld), requiredDeposit: round2(requiredDeposit), difference: round2(difference), psdrSavings: hasSavings ? round2(difference) : 0, recommendation: hasSavings ? "Apply for Permanent Security Deposit Refund (PSDR)." : "No PSDR savings opportunity found.", calculationBasis, }; } function round2(num) { return Math.round(num * 100) / 100; } /** * Formats the result to match your specified output block. */ function formatPSDROutput(result) { if (result.error) { return `PSDR Analysis Error: ${result.error}`; } const lines = [ `Supply Type:`, `${result.supplyType}`, ``, `Security Deposit Held:`, `₹${result.securityDepositHeld}`, ``, `Y (Required Security Deposit):`, `₹${result.requiredDeposit}`, ``, `PSDR Savings:`, `₹${result.psdrSavings}`, ``, `Recommendation:`, result.psdrSavings > 0 ? `Apply for Permanent Security Deposit Refund (PSDR).\n\nEstimated Refund = ₹${result.psdrSavings}` : `No PSDR savings opportunity found.`, ]; return lines.join("\n"); } /** * Wires a PSDR result into the shared recommendation-engine format used by * the CD and PF savings modules: * Observation → Root Cause → Recommended Action → Expected ₹ Impact → * Confidence → Urgency * * Returns null when there's no PSDR opportunity, so it can be filtered out * of the "active recommendations" list the same way a null/zero CD or PF * recommendation would be. * * Note: unlike CD/PF, PSDR is a one-time refund, not a recurring monthly * saving — flag it as such (isRecurring: false) so it's kept in a separate * bucket from your annualized savings total in reports/dashboards. */ function buildPSDRRecommendation(billData) { const result = calculatePSDR(billData); if (!result.psdrSavings || result.psdrSavings <= 0) { return null; } const { supplyType, securityDepositHeld, requiredDeposit, calculationBasis } = result; const basisText = supplyType === "HT" ? `required deposit is based on 2× the greater of "If Paid Upto" (₹${calculationBasis.ifPaidUpto}) and "Last Month Payment" (₹${calculationBasis.lastMonthPayment})` : `required deposit is based on 2× the average bill amount over the last ${calculationBasis.monthsUsed} month(s) of Billing History (₹${calculationBasis.averageBillAmount}/month avg)`; return { service: "PSDR", isRecurring: false, // one-time refund — exclude from annualized savings totals observation: `Security Deposit Held (₹${securityDepositHeld}) exceeds the required deposit of ₹${requiredDeposit} for this ${supplyType} connection.`, rootCause: `Under MSEDCL norms, the ${basisText}. The deposit currently held is higher than this requirement, indicating an excess balance.`, recommendedAction: "Apply for a Permanent Security Deposit Refund (PSDR) with MSEDCL to recover the excess amount held.", expectedImpact: { amount: result.psdrSavings, label: `₹${result.psdrSavings} one-time refund`, }, confidence: "High", // deterministic calculation from bill-stated figures urgency: "Low", // no penalty accrues from delay, but capital sits idle until claimed calculationBasis, }; } /** * Convenience: merge PSDR alongside CD/PF recommendation builders. * Assumes buildCDRecommendation(billData) and buildPFRecommendation(billData) * already exist in your codebase and follow the same shape * (service, isRecurring, observation, rootCause, recommendedAction, * expectedImpact, confidence, urgency). * * Example: * * function buildAllRecommendations(billData) { * return [ * buildCDRecommendation(billData), * buildPFRecommendation(billData), * buildPSDRRecommendation(billData), * ].filter(Boolean); // drop nulls where no opportunity exists * } */ // Example usage: // // const htBill = { // supplyType: "HT", // ifPaidUpto: 450000, // lastMonthPayment: 470000, // securityDepositHeld: 1200000 // }; // console.log(formatPSDROutput(calculatePSDR(htBill))); // // const ltBill = { // supplyType: "LT", // billingHistory: [ // { billAmount: 85000 }, // { billAmount: 92000 }, // { billAmount: 78000 }, // ], // securityDepositHeld: 250000 // }; // console.log(formatPSDROutput(calculatePSDR(ltBill))); export { calculatePSDR, formatPSDROutput, buildPSDRRecommendation };
